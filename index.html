⚙ Google Sheets Setup

This app can submit match data directly to your Google Sheet using a Google Apps Script web app. One-time setup:

Step 1 — Open your Google Sheet

Go to Extensions → Apps Script and replace all the code with:

function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // Match summary sheet var summary = ss.getSheetByName("Match Summary"); if (!summary) { summary = ss.insertSheet("Match Summary"); summary.appendRow(["Match #","Date","Routine","Auto Fuel","Total Cycles","SOM Cycles", "Avg Cycle (s)","Best Cycle (s)","Avg Transit (s)","Total Intake (s)", "Total Score (s)","Total Feed (s)","Total Transit (s)","Notes"]); } summary.appendRow([ data.matchNum, data.date, data.routine, data.autoFuel, data.cycles, data.somCycles||0, data.avgCycle, data.bestCycle, data.avgTransit, data.totalIntake, data.totalScore, data.totalFeed||0, data.totalTransit, data.notes ]); // Event detail sheet var detail = ss.getSheetByName("Event Detail"); if (!detail) { detail = ss.insertSheet("Event Detail"); detail.appendRow(["Match #","Date","Routine","Phase","Event","Start (s)","End (s)","Duration (s)"]); } data.events.forEach(function(ev) { detail.appendRow([data.matchNum, data.date, data.routine, ev.phase, ev.type, ev.start.toFixed(2), ev.end.toFixed(2), ev.duration.toFixed(2)]); }); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error", message: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
